                        $("#divAlert").show();

                return false;

            if (!confirmADPassword(newPass, confirmNewPass)) {
                return false;
            }
            var jsonObject = {
                currntADPassword: $("#txtCurrentPass").val(),
                newADPassword: $("#txtNewPass").val()

                }
            });
